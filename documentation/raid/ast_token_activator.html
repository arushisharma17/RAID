 50        """Recursively extracts tokens from the leaf nodes of the AST, including depth."""
 51        if node is None:

 53        tokens = []

 92    def __init__(self, model_name, device='cpu', binary_filter='set:public,static', output_prefix='output', aggregation_method='mean'):
 93        self.model_name = model_name
 94        self.device = device
 95        self.binary_filter = binary_filter
 96        self.output_prefix = output_prefix
 97        self.binary_filter_compiled = None
 98        self.aggregation_method = aggregation_method

100    def process_activations(self, tokens_tuples, output_dir):
101        """Generates activations, parses them, handles binary filter, and annotates data."""
102        input_file = os.path.join(output_dir, 'input_sentences.txt')
103        output_file = os.path.join(output_dir, 'activations.json')
104        # Generate activations
105        self.generate_activations(input_file, output_file)
106        # Parse activations
107        extracted_tokens, activations = self.parse_activations(output_file)
108        # Handle binary filter
109        self.handle_binary_filter()
110        # Prepare tokens with depth
111        tokens_with_depth = [(t, d) for (_, t, d) in tokens_tuples]
112        # Annotate data
113        self.annotate_data(tokens_with_depth, activations, output_dir)
114        # Write aggregated activations
115        self.write_aggregated_activations(tokens_with_depth, activations, output_dir)
116        # Aggregate phrase activations
117        phrase_activations = self.aggregate_phrase_activations(tokens_with_depth, activations, method=self.aggregation_method)
118        # Output mapping to file
119        self.write_phrase_activations(phrase_activations, output_dir)

121    def generate_activations(self, input_file, output_file):
122        """Generates activations using the specified transformer model."""
123        transformers_extractor.extract_representations(
124            self.model_name,
125            input_file,
126            output_file,
127            aggregation="average",  
128            output_type="json",
129            device=self.device
130        )

132    def parse_activations(self, activation_file):
133        """Parses the activations from the JSON output."""
134        with open(activation_file, 'r', encoding='utf-8') as f:
135            activation_data = json.load(f)
136        activations = []
137        extracted_tokens = []
138        for feature in activation_data['features']:
139            token = feature['token'].replace('Ġ', '')
140            layers = feature['layers']
141            # Collect activations from all layers
142            token_activations = []
143            for layer in layers:
144                layer_index = layer['index']
145                layer_values = layer['values']
146                token_activations.append((layer_index, np.array(layer_values)))
147            activations.append(token_activations)
148            extracted_tokens.append(token)
149        return extracted_tokens, activations
150
151    def handle_binary_filter(self):
152        """Compiles the binary filter based on user input."""
153        if self.binary_filter.startswith("re:"):
154            self.binary_filter_compiled = re.compile(self.binary_filter[3:])
155        elif self.binary_filter.startswith("set:"):
156            self.binary_filter_compiled = set(self.binary_filter[4:].split(","))
157        else:
158            raise NotImplementedError("Filter must start with 're:' for regex or 'set:' for a set of words.")
159
160    def annotate_data(self, tokens_with_depth, activations, output_dir):
161        """Creates binary data and saves it with tokens and labels organized by AST depth."""
162        tokens_depths, labels, flat_activations = self._create_binary_data(
163            tokens_with_depth, activations, self.binary_filter_compiled, balance_data=False
164        )
165
166        # Collect tokens and labels by depth
167        from collections import defaultdict
168        depth_to_tokens = defaultdict(list)
169        depth_to_labels = defaultdict(list)
170
171        for (token, depth), label in zip(tokens_depths, labels):
172            depth_to_tokens[depth].append(token)
173            depth_to_labels[depth].append(label)
174
175        # Determine the maximum depth
176        max_depth = max(depth_to_tokens.keys())
177
178        # Save the files
179        words_file = os.path.join(output_dir, f"{self.output_prefix}_tokens.txt")
180        labels_file = os.path.join(output_dir, f"{self.output_prefix}_labels.txt")
181        activations_file = os.path.join(output_dir, f"{self.output_prefix}_activations.txt")
182
183        with open(words_file, "w", encoding='utf-8') as f_words, open(labels_file, "w", encoding='utf-8') as f_labels:
184            # Iterate over all depths from 0 to max_depth
185            for depth in range(max_depth + 1):
186                tokens_line = ' '.join(depth_to_tokens.get(depth, []))
187                labels_line = ' '.join(depth_to_labels.get(depth, []))
188                f_words.write(tokens_line + '\n')
189                f_labels.write(labels_line + '\n')
190
191        # Save activations to text file (order maintained as per tokens_with_depth)
192        # Since activations now contain per-layer activations, we can save only the last layer if needed
193        with open(activations_file, 'w', encoding='utf-8') as f:
194            for token_activations in flat_activations:
195                # Get activation from the last layer
196                last_layer_activation = token_activations[-1][1].tolist()
197                activation_str = ' '.join(map(str, last_layer_activation))
198                f.write(activation_str + '\n')
199
200        print(f"Words saved to '{words_file}'.")
201        print(f"Labels saved to '{labels_file}'.")
202        print(f"Activations saved to '{activations_file}'.")

204    def _create_binary_data(self, tokens_with_depth, activations, binary_filter, balance_data=False):
205        """Creates a binary labeled dataset based on the binary_filter."""
206        if isinstance(binary_filter, set):
207            filter_fn = lambda x: x in binary_filter
208        elif isinstance(binary_filter, Pattern):
209            filter_fn = lambda x: binary_filter.match(x)
210        elif callable(binary_filter):
211            filter_fn = binary_filter
212        else:
213            raise NotImplementedError("ERROR: The binary_filter must be a set, a regex pattern, or a callable function.")
214
215        words = []
216        depths = []
217        labels = []
218        final_activations = []
219
220        for (token, depth), token_activations in zip(tokens_with_depth, activations):
221            words.append(token)
222            depths.append(depth)
223            if filter_fn(token):
224                labels.append('positive')
225            else:
226                labels.append('negative')
227            final_activations.append(token_activations)
228
229        return list(zip(words, depths)), labels, final_activations

231    def aggregate_activation_list(self, activations, method='mean'):
232        """
233        Aggregates a list of activations using the specified method.
234        """
235        activations_array = np.stack(activations)
236        if method == 'mean':
237            return np.mean(activations_array, axis=0)
238        elif method == 'max':
239            return np.max(activations_array, axis=0)
240        elif method == 'sum':
241            return np.sum(activations_array, axis=0)
242        elif method == 'concat':
243            return np.concatenate(activations_array, axis=0)
244        else:
245            raise ValueError("Unsupported aggregation method")
246
247    def aggregate_phrase_activations(self, tokens_with_depth, activations, method='mean'):
248        """
249        Aggregates token-level activations into phrase-level activations.
250        Phrases are defined as tokens at the same depth level.
251        """
252        from collections import defaultdict
253
254        # Group tokens and activations by depth (phrases)
255        depth_to_tokens = defaultdict(list)
256        depth_to_activations = defaultdict(list)
257        for (token, depth), token_layers in zip(tokens_with_depth, activations):
258            depth_to_tokens[depth].append(token)
259            depth_to_activations[depth].append(token_layers)  # token_layers is list of (layer_index, activation)
260
261        # For each depth (phrase), aggregate activations per layer
262        phrase_activations = []
263        for depth in sorted(depth_to_tokens.keys()):
264            tokens = depth_to_tokens[depth]
265            tokens_layers_list = depth_to_activations[depth]
266
267            # Aggregate activations per layer index
268            layer_to_activations = defaultdict(list)
269            for token_layers in tokens_layers_list:
270                for layer_index, activation in token_layers:
271                    layer_to_activations[layer_index].append(activation)
272
273            # Prepare aggregated layers
274            aggregated_layers = []
275            for layer_index in sorted(layer_to_activations.keys()):
276                activations_list = layer_to_activations[layer_index]
277                # Aggregate activations using the specified method
278                aggregated_activation = self.aggregate_activation_list(activations_list, method=method)
279                aggregated_layers.append({
280                    "index": layer_index,
281                    "values": aggregated_activation.tolist()
282                })
283
284            # Use the tokens as a phrase key (joined tokens)
285            phrase_key = ' '.join(tokens)
286            phrase_feature = {
287                "phrase": phrase_key,
288                "layers": aggregated_layers
289            }
290            phrase_activations.append(phrase_feature)
291
292        # Build the output data structure
293        output_data = {
294            "linex_index": 0,
295            "features": phrase_activations
296        }
297
298        return output_data
299
300    def write_phrase_activations(self, phrase_activations, output_dir):
301        """Writes the phrase activations to a JSON file in the same format as activations.json."""
302        # Construct output file path
303        mapping_file = os.path.join(output_dir, f"{self.output_prefix}_phrasal_activations.json")
304        # Write to file
305        with open(mapping_file, 'w', encoding='utf-8') as f:
306            json.dump(phrase_activations, f, indent=2)
307        print(f"Phrase activations saved to '{mapping_file}'.")
308
309    def write_aggregated_activations(self, tokens_with_depth, activations, output_dir):
310        """Writes the mean-aggregated activations across all layers to a JSON file."""
311        aggregated_file = os.path.join(output_dir, f"{self.output_prefix}_aggregated_activations.json")
312        
313        # Create a list to store token-activation mappings
314        token_activations = []
315        
316        # Iterate through tokens and their activations together
317        for (token, _), token_layers in zip(tokens_with_depth, activations):
318            # Get activations from all layers for this token
319            layer_activations = [layer[1] for layer in token_layers]
320            # Aggregate across layers using the specified method
321            aggregated = self.aggregate_activation_list(layer_activations, method=self.aggregation_method)
322            
323            # Create feature object for this token
324            token_feature = {
325                "token": token,  # Use actual token from tokens_with_depth
326                "aggregated_values": aggregated.tolist()
327            }
328            token_activations.append(token_feature)
329        
330        # Build the output data structure
331        output_data = {
332            "aggregation_method": self.aggregation_method,
333            "features": token_activations
334        }
335        
336        # Write to JSON file
337        with open(aggregated_file, 'w', encoding='utf-8') as f:
338            json.dump(output_data, f, indent=2)
339        
340        print(f"Aggregated activations saved to '{aggregated_file}'.")

205    def _create_binary_data(self, tokens_with_depth, activations, binary_filter, balance_data=False):
206        """Creates a binary labeled dataset based on the binary_filter."""
207        if isinstance(binary_filter, set):
208            filter_fn = lambda x: x in binary_filter
209        elif isinstance(binary_filter, Pattern):
210            filter_fn = lambda x: binary_filter.match(x)
211        elif callable(binary_filter):
212            filter_fn = binary_filter
213        else:
214            raise NotImplementedError("ERROR: The binary_filter must be a set, a regex pattern, or a callable function.")
215
216        words = []
217        depths = []
218        labels = []
219        final_activations = []
220
221        for (token, depth), token_activations in zip(tokens_with_depth, activations):
222            words.append(token)
223            depths.append(depth)
224            if filter_fn(token):
225                labels.append('positive')
226            else:
227                labels.append('negative')
228            final_activations.append(token_activations)
229
230        return list(zip(words, depths)), labels, final_activations

233        """

249        """

251        Phrases are defined as tokens at the same depth level.
252        """
253        from collections import defaultdict

255        # Group tokens and activations by depth (phrases)

263        phrase_activations = []
264        for depth in sorted(depth_to_tokens.keys()):

266            tokens_layers_list = depth_to_activations[depth]

268            # Aggregate activations per layer index
269            layer_to_activations = defaultdict(list)
270            for token_layers in tokens_layers_list:

274            # Prepare aggregated layers
275            aggregated_layers = []

282                    "values": aggregated_activation.tolist()
283                })

286            phrase_key = ' '.join(tokens)
287            phrase_feature = {
288                "phrase": phrase_key,
289                "layers": aggregated_layers
290            }
291            phrase_activations.append(phrase_feature)

293        # Build the output data structure
294        output_data = {
